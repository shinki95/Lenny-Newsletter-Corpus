this.getKeywordList = (quotationChoiceList) => {
return quotationChoiceList.reduce(
(result, quotationChoice) => [
...result,
...quotationChoice.keywords.map((keyword) => ({
choiceId: quotationChoice.choice_id,
bridgeUrl: quotationChoice.bridge_url,
title: quotationChoice.title,
keyword,
})),
],
[],
)
}

const searchQuery = this.$desktopSearchInput.value
if (!searchQuery) {
return
}

this.$desktopSearchResultWindow.classList.remove('hide')

this.renderEmptyResult = function (target) {
this.$isResultEmpty = true

target.innerHTML = `
<div class="empty-result-wrapper">
<div class='empty-result'>검색 결과가 없어요</div>
</div>
`

this.renderSearchResult = lodash.debounce(function (searchQuery, target) {
if (!searchQuery) {
this.hideClearIcon()
this.renderEmptyResult(target)
return
}

this.showClearIcon()
const choiceSearchResult = this.quotationChoiceList.filter((keyword) => searchQuery !== '' && keyword.keyword.includes(searchQuery))

axiosGET('/api/v1/quotation/sub-choice/search', { word: searchQuery })
.then(res => {
this.$isResultEmpty = Boolean(res.data.length)

const subChoiceSearchResult = this.getKeywordList(res.data).filter((keyword) => searchQuery !== '' && keyword.keyword.includes(searchQuery))
if (choiceSearchResult.length + subChoiceSearchResult.length === 0) {
this.renderEmptyResult(target)
return
}

target.innerHTML = `
${subChoiceSearchResult.length === 0 ? '' : (`
<div class="result-wrapper">
<div class='title'>의뢰서 작성</div>
<ul>
${subChoiceSearchResult.map(choice => `
<li>
<a href="/quotation/question/${choice.choiceId}?breadcrumbs=${choice.title}" class="result-item">
<div class="keyword">${choice.keyword.replaceAll(searchQuery, `<b>${searchQuery}</b>`)}</div>
<div class="arrow">
<svg width="24" height="24" viewBox="0 0 24 24" fill="none" xmlns="http://www.w3.org/2000/svg">
<path d="M10 8L14 11.9999L10 16" stroke="#AEB5BA" />
</svg>
</div>
<div class="category">${choice.title}</div>
</a>
</li>
`).join('')}
</ul>
</div>
`)}
${choiceSearchResult.length === 0 ? '' : (`
<div class="result-wrapper">
<div class='title'>분야 홈</div>
<ul>
${choiceSearchResult.map(choice => `
<li>
<a href="${choice.bridgeUrl}" class="result-item">
<div class="keyword">${choice.keyword.replaceAll(searchQuery, `<b>${searchQuery}</b>`)}</div>
<div class="arrow">
<svg width="24" height="24" viewBox="0 0 24 24" fill="none" xmlns="http://www.w3.org/2000/svg">
<path d="M10 8L14 11.9999L10 16" stroke="#AEB5BA" />
</svg>
</div>
<div class="category">${choice.title}</div>
</a>
</li>
`).join('')}
</ul>
</div>
`)}
`
})
.catch(err => {
this.hideClearIcon()
this.renderEmptyResult(target)
})
}, 300)

switch (index) {
case 0:
show(this.$desktopDropBoxFirst)
show(this.$desktopTransparentFirst)
break
case 1:
show(this.$desktopDropBoxSecond)
show(this.$desktopTransparentSecond)

}

case 0:
hide(this.$desktopDropBoxFirst)
hide(this.$desktopTransparentFirst)
break
case 1:
hide(this.$desktopDropBoxSecond)
hide(this.$desktopTransparentSecond)
break
}
}
}

}

show(this.$mobileBlackWindow)
} else {

hide(this.$mobileInnerNav)
hide(this.$mobileBlackWindow)
}

}
}

const regex = /^\/boards\/\d+$/
if (regex.test(location.pathname))
return

}

}

}

for (let i = 0; i < this.$postCountList.length; i++) {
show(this.$postCountList[i])
this.$postCountList[i].innerText = this.postCount
}
}

}
